%------------------------------------------------------------------ Header con Foto CIRCULAR
\newcommand{\simpleheader}[5]{
  \begin{tikzpicture}[remember picture,overlay]
    % Fondo de color del encabezado
    \node [rectangle, fill=#1, anchor=north, minimum width=\paperwidth, minimum height=3.5cm] (box) at (current page.north){};
    
    % NUEVO: Foto de perfil CIRCULAR con borde blanco elegante
    \node [
        circle,              % Forma circular
        %draw=,           % Color del borde
       % line width=0pt,       % Grosor del borde
        minimum size=3cm,   % Tamaño total del círculo visible
        anchor=west,
        xshift=2cm,           % Separación del borde izquierdo
        path picture={        % Esta es la magia: recorta la imagen dentro del círculo
            \node [anchor=center] at (path picture bounding box.center) {
                % Asegurate que la altura/ancho sea suficiente para llenar el círculo
                \includegraphics[height=6.5cm, keepaspectratio]{perfilcami.jpg}
            };
        }
    ] at (box.west) {};

    % Nombre y Apellido (centrados en la franja)
    \node [anchor=center] (name) at (box) {%
      \fontsize{35pt}{40pt}\selectfont \color{#5} #2 \textbf{#3}
    };
    % Título/Puesto
    \node [anchor=north] at (name.south) {%
      \fontsize{15pt}{18pt}\selectfont \color{#5} #4
    };
  \end{tikzpicture}
  \vspace{2.5cm} % Espacio debajo del header para que no se superponga el texto
}
